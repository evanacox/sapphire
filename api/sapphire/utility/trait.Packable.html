Expand description

Helper trait for a type that can be packed into a PackedOption.

These types need to have some null-ish value that they can reserve, that value will be used to distinguish between None and Some.

struct NonZero(i32);

impl Packable for NonZero {
    fn reserved_null() -> Self {
        NonZero(0)
    }

    fn is_reserved_null(&self) -> bool {
        self.0 == 0
    }
}

let opt = PackedOption::some(NonZero(15));

assert_eq!(opt.is_some(), true);
